        console.log("Component Did Update");
    }

    componentWillUnmount(){
        console.log("Component will unmount");
    }

        // const {count, count1} = this.state;

        // console.log(this.props.name + "Child Render Called");

        const {name, location, login, avatar_url} = this.state.userInfo;

        return (
            <div className="user-card">

React Lifecycle Cheat Sheet (Class vs Functional)

Think of React components like a human life cycle:

Birth → Life → Update → Death

In React:

Mounting → Updating → Unmounting
1️⃣ Mounting Phase (Component is Created)

This phase happens when the component appears on the screen for the first time.

Class Component Lifecycle

Order:

constructor()
   ↓
render()
   ↓
componentDidMount()

Example from your code (UserClass.js):

 super(props);

 this.state = {
   userInfo:{
     name:"Dummy",
     location:"Default"
   }
 }
}

Purpose:

Initialize state
Access props

Example:

name = Dummy
location = Default
render()
render(){
 return(
  <div>
   <h4>Name: {name}</h4>
  </div>
 )
}

Purpose:

Create UI
Return JSX

React converts JSX into DOM elements.

componentDidMount()
async componentDidMount(){
 const data = await fetch("API");
 const json = await data.json();

 this.setState({
  userInfo: json
 });
}

Purpose:

API calls
Data fetching
Subscriptions

This runs after UI appears.
